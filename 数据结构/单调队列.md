## 单调队列
### 滑动窗口
```cpp
vector<int> ans;
deque<int> dq;

// 宽度为 k 的窗口中的最大值
for(int i=1;i<=n;++i)
{
    // 保证区间内数均在窗口中
    while(dq.size() && dq.front() <= i-k)	dq.pop_front();
    // 保证窗口中元素单调
    while(dq.size() && a[dq.back()] <= a[i])	dq.pop_back();
    dq.push_back(i);
    if(i >= k)	ans.push_back(a[dq.front()]);
}
// 宽度为 k 的窗口中的最小值
for(int i=1;i<=n;++i)
{
    while(dq.size() && dq.front() <= i-k)	dq.pop_front();
    while(dq.size() && a[dq.back()] <= a[i])	dq.pop_back();
    dq.push_back(i);
    if(i >= k)	ans.push_back(a[dq.front()]);
}
```
