## 最短路
### dijkstra
```cpp
void dijkstra(int st,int d[])
{
	for(int i=1;i<=n;++i)	d[i] = M;// 初始化
	priority_queue<edge> pq;
	d[st] = 0;// 起始位置
	bitset<N> vis;// 去重，防止重复访问降速
	pq.push({st,d[st]});
	while(pq.size())
	{
		int x = pq.top().y;	pq.pop();
		if(vis[x])	continue;
		vis[x] = true;
		for(auto &[y,v] : a[x])
		{
			if(d[y] > d[x] + v)
			{
				d[y] = d[x] + v;
				pq.push({y,d[y]});
			}
		}
	}
}
```

### SPFA
```cpp
bool spfa(int st,int d[])
{
	queue<int> q;
	bitset<N> inq;
	q.push(st);
	vector<int> cnt(n+1);
	while(q.size())
	{
		int x = q.front();	q.pop();	inq[x] = false;
		for(auto &[y,v] : a[x])
		{
			if(d[y] > d[x] + v)
			{
				if(++ cnt[y] > n)	return true; // 负环
				d[y] = d[x] + v;
				if(!inq[y])	q.push(y), inq[y] = true;
			}
		}
	}
	return false;
}
```
